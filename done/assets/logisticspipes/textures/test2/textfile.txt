    list.modid = list.filename.split(Regex("-"))[0]
    list.version = list.filename.removePrefix(list.modid + "-").removeSuffix(".jar")

        when {
            name.isPng() -> list.files.add(name)
            name.isDir() && name.hasSuffix("assets/") && name.hasMatching("textures")-> list.dirs.add(name)
        }
